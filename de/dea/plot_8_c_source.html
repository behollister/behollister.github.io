338 void trajectory_step (double dt)
339 {
340  static int i = 0;
341  int n = 1 + Sys->parts.size(), m = Anim->size();
342  double t = AnimTime + dt;
343  if (t < 0.0) {
344  t = Anim->at(m-n);
345  i = m - 2*n;
346  }
347  else if (t >= Anim->at(m-n)) {
348  t = 0.0;
349  i = 0;
350  }
351  else if (dt > 0.0)
352  while (i < m - n && t > Anim->at(i+n))
353  i += n;
354  else
355  while (i > 0 && t < Anim->at(i))
356  i -= n;
357  AnimTime = t;
358  double mu = (t - Anim->at(i))/(Anim->at(i+n) - Anim->at(i));
359  for (int j = 0; j < n - 1; ++j)
360  Config[j] = (1.0 - mu)*Anim->at(i+j+1) + mu*Anim->at(i+j+1+n);
361 }
